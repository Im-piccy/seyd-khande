cmake_minimum_required(VERSION 3.10)
project(seyd_khande)
add_executable(app src/main.cpp src/game.cpp src/user.cpp src/controller.cpp)
target_include_directories(app PRIVATE raylib/include)
target_link_libraries(app PRIVATE ${CMAKE_SOURCE_DIR}/raylib/lib/libraylib.a opengl32 gdi32 winmm)
set_target_properties(app PROPERTIES RUNTIME_OUTPUT_DIRECTORY "${CMAKE_SOURCE_DIR}")
